class Solution(object):
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """

        
        for x in range(len(nums)):
            for y in range(x+1, len(nums)):
                two_sum = nums[x] + nums[y]
                if two_sum == target:
                    return [x,y]
